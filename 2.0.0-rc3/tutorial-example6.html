        var checkedItems = [];
        var $removeButton = $('.tui-btn-cancel');

        var uploader = new tui.component.FileUploader($('#uploader'), {
            url: {
                send: 'http://localhost:3000/upload'
            },
            isMultiple: true,
            isBatchTransfer: true,
            listUI: {
                type: 'table'
            }
        });

        uploader.on('check', function(evt) {
            if (evt.state) {
                checkedItems.push(evt);
            } else {
                checkedItems = tui.util.filter(checkedItems, function(item) {
                    return !(item.id === evt.id);
                });
            }
        });

        uploader.on('checkAll', function(evt) {
            checkedItems = evt.filelist;
        });

        $removeButton.on('click', function() {
            uploader.removeList(checkedItems);
        });
